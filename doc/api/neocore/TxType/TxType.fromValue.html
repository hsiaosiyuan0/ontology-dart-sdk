factory TxType.fromValue(int value) {
  switch (value) {
    case 0x02:
      return bookKeeper;
    case 0x03:
      return claim;
    case 0xd0:
      return deploy;
    case 0xd1:
      return invoke;
    case 0x04:
      return enrollment;
    case 0x05:
      return vote;
    default:
      throw ArgumentError('Invalid label');
  }
}
