dynamic pushInt (int x)

Implementation 

pushInt(int x) {
  if (x == -1) {
    pushNum(OpCode.pushm1);
  } else if (x == 0) {
    pushNum(OpCode.push0);
  } else if (x > 0 && x < 16) {
    pushNum(OpCode.push1 - 1 + x);
  } else {
    pushBigInt(BigInt.from(x));
  }
}
