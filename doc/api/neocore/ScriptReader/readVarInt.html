int readVarInt() {
  int len = readUint8();
  if (len == 0xfd) {
    return readUint16LE();
  } else if (len == 0xfe) {
    return readUint32LE();
  } else if (len == 0xff) {
    return readUint64LE();
  }
  return len;
}
