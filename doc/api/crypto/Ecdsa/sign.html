static Future<Uint8List> sign(
    Uint8List msg, Uint8List pri, Curve curve, int sigSchemaVal) async {
  var res =
      await invokeCrypto('ecdsa.sig', [msg, pri, curve.label, sigSchemaVal]);
  return res as Uint8List;
}
